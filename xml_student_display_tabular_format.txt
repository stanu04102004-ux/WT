<?php
if(isset($_GET['course'])) {
    $xml = simplexml_load_file("student.xml");
    $course = $_GET['course'];

    echo "<table border=1>";
    foreach($xml->student as $s) {
        if($s->course == $course) {
            echo "<tr>
                <td>$s->rollno</td>
                <td>$s->name</td>
                <td>$s->address</td>
                <td>$s->college</td>
            </tr>";
        }
    }
    echo "</table>";
}
