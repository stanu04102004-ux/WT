then open controller vim Stud_controller.php
<?php
defined('BASEPATH') OR exit('No direct script access allowed');

class Stud_controller extends CI_Controller
 {
	public function _construct()
	{
		parent:: _construct();
		$this->load->model('Student_Model');
	}

	
	public function index()
	{
		$this->load->view('Student_view');
	}
	public function insert_data()
	{
		$data=array('roll'=>$this->input->post('roll'),'name'=>$this->input->post('name'),'class'=>$this->input->post('class'));
		$this->load->model('Student_Model');
		$id=$this->Student_Model->insert_user($data);
		//echo "Id is $id";
		if($id==1)
			echo "Record inserted successfully";
		else
			echo "Record not inserted";
		redirect('Stud_controller');
	}
}
?>

then open model Student_Model.php
<?php
class Student_model extends CI_Model 
{
	public function insert_user($data)
	{
		$this->load->database();
		return $this->db->insert('student',$data);
	}
}
?>
