<?php
session_start();

if (!isset($_SESSION['attempt'])) {
    $_SESSION['attempt'] = 0;
}

$correct_user = "admin";
$correct_pass = "123";
?>
<!DOCTYPE html>
<html>
<body>

<form method="post">
Username: <input type="text" name="user"><br>
Password: <input type="password" name="pass"><br>
<input type="submit" name="login">
</form>

<?php
if(isset($_POST['login'])) {
    if($_POST['user'] == $correct_user && $_POST['pass'] == $correct_pass) {
        echo "<h3>Welcome</h3>";
        $_SESSION['attempt'] = 0;
    } else {
        $_SESSION['attempt']++;
        if($_SESSION['attempt'] >= 3) {
            echo "Access Denied";
        } else {
            echo "Wrong! Remaining: " . (3 - $_SESSION['attempt']);
        }
    }
}
?>

</body>
</html>
